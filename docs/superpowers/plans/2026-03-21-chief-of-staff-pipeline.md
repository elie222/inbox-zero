# Chief of Staff Pipeline Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a Chief of Staff email processing pipeline to a fork of Inbox Zero that categorizes emails with Claude, integrates with Acuity Scheduling, cross-references Google Calendar with prefix conventions, and posts to Slack for approve/edit/reject approval.

**Architecture:** Additive layer on Inbox Zero — a separate Pub/Sub subscription delivers Gmail notifications to `/api/chief-of-staff/webhook`, which runs a pre-filter (blocks junk without Claude), then sends important emails to Claude with custom tools (Acuity, Calendar). Results post to Slack #chief-of-staff with Block Kit buttons. Existing Inbox Zero features remain untouched.

**Tech Stack:** Next.js (App Router), Prisma + Postgres, Vercel AI SDK (`ai` package) + Zod, `@slack/web-api`, `@googleapis/gmail` + `@googleapis/calendar`, Vitest for testing.

**Design Spec:** `docs/superpowers/specs/2026-03-21-chief-of-staff-pipeline-design.md`

---

## Prerequisites

Before starting implementation:

1. Fork `elie222/inbox-zero` to your own repo
2. Clone the fork locally
3. Run `pnpm install` and verify the app builds (`pnpm build`)
4. Set up Postgres locally (Docker: `docker-compose up -d`)
5. Copy `.env.example` to `.env` and fill in required values
6. Run `npx prisma migrate dev` to initialize the database

The plan assumes the forked repo is your working directory. All paths are relative to the repo root.

```typescript
// apps/web/utils/chief-of-staff/acuity/client.ts
const ACUITY_BASE_URL = "https://acuityscheduling.com/api/v1";
const MAX_RETRIES = 3;
const RETRY_DELAYS = [1000, 2000, 4000]; // ms

export class AcuityApiError extends Error {
  constructor(
    public status: number,
    public statusText: string,
    public body: string,
  ) {
    super(`Acuity API error: ${status} ${statusText}`);
    this.name = "AcuityApiError";
  }
}

function getAuthHeader(): string {
  const userId = process.env.ACUITY_USER_ID;
  const apiKey = process.env.ACUITY_API_KEY;
  if (!userId || !apiKey) {
    throw new Error("ACUITY_USER_ID and ACUITY_API_KEY must be set");
  }
  return `Basic ${Buffer.from(`${userId}:${apiKey}`).toString("base64")}`;
}

export async function acuityFetch<T = unknown>(
  method: "GET" | "POST" | "PUT" | "DELETE",
  path: string,
  body?: Record<string, unknown>,
): Promise<T> {
  const url = `${ACUITY_BASE_URL}${path}`;
  const headers: Record<string, string> = {
    Authorization: getAuthHeader(),
    "Content-Type": "application/json",
  };

  for (let attempt = 0; attempt <= MAX_RETRIES; attempt++) {
    const response = await fetch(url, {
      method,
      headers,
      body: body ? JSON.stringify(body) : undefined,
    });

    if (response.ok) {
      return response.json() as Promise<T>;
    }

    if (response.status === 429 && attempt < MAX_RETRIES) {
      await new Promise((resolve) =>
        setTimeout(resolve, RETRY_DELAYS[attempt]),
      );
      continue;
    }

    const errorBody = await response.text();
    throw new AcuityApiError(response.status, response.statusText, errorBody);
  }

  // Unreachable, but TypeScript needs it
  throw new Error("Max retries exceeded");
}
```

```typescript
// apps/web/utils/chief-of-staff/calendar/prefix-parser.ts

export enum PrefixType {
  HARD_BLOCK = "hard_block",
  SOFT = "soft",
  INFORMATIONAL = "informational",
}

export interface ParsedPrefix {
  type: PrefixType;
  cleanTitle: string;
}

export function parseEventPrefix(title: string): ParsedPrefix {
  const trimmed = title.trim();

  if (trimmed.startsWith("~")) {
    return {
      type: PrefixType.SOFT,
      cleanTitle: trimmed.slice(1).trim(),
    };
  }

  const fyiMatch = trimmed.match(/^fyi:\s*/i);
  if (fyiMatch) {
    return {
      type: PrefixType.INFORMATIONAL,
      cleanTitle: trimmed.slice(fyiMatch[0].length).trim(),
    };
  }

  return {
    type: PrefixType.HARD_BLOCK,
    cleanTitle: trimmed,
  };
}
```

```typescript
import { TIMEZONE } from "../types";

interface DayProtectedResult {
  protected: boolean;
  reason?: string;
  overridable?: boolean;
}

export function isDayProtected(
  date: Date,
  isVip: boolean,
): DayProtectedResult {
  // Get day of week in Chicago timezone
  const dayOfWeek = new Intl.DateTimeFormat("en-US", {
    weekday: "short",
    timeZone: TIMEZONE,
  }).format(date);

  if (dayOfWeek === "Tue") {
    return {
      protected: true,
      reason: "Tuesday is a protected recovery day",
      overridable: false,
    };
  }

  if (dayOfWeek === "Fri" && !isVip) {
    return {
      protected: true,
      reason: "Friday is a protected non-tutoring day",
      overridable: true,
    };
  }

  return { protected: false };
}
```

```typescript
// apps/web/utils/chief-of-staff/calendar/checker.ts
import { calendar_v3 } from "@googleapis/calendar";
import { parseEventPrefix, PrefixType } from "./prefix-parser";
import { isDayProtected } from "./day-protection";
import {
  CalendarAvailability,
  CalendarConflict,
  CALENDAR_IDS,
  SOFT_CALENDARS,
  TIMEZONE,
} from "../types";

interface CheckCalendarParams {
  startTime: Date;
  endTime: Date;
  isVip: boolean;
  calendarAuth: { accessToken: string; refreshToken: string };
}

const BUFFER_MS = 15 * 60 * 1000; // 15 minutes

const CALENDAR_NAMES: Record<string, string> = {
  [CALENDAR_IDS.personal]: "Personal",
  [CALENDAR_IDS.smartCollege]: "Smart College",
  [CALENDAR_IDS.rmsWork]: "RMS Work",
  [CALENDAR_IDS.praxis]: "Praxis",
  [CALENDAR_IDS.nutrition]: "Nutrition",
  [CALENDAR_IDS.workout]: "Workout",
};

export async function checkCalendarAvailability(
  params: CheckCalendarParams,
): Promise<CalendarAvailability> {
  const { startTime, endTime, isVip } = params;

  // Check day protection first
  const dayCheck = isDayProtected(startTime, isVip);
  if (dayCheck.protected) {
    return {
      available: false,
      hardBlocks: [
        {
          title: dayCheck.reason!,
          calendar: "Day Protection",
          start: startTime.toISOString(),
          end: endTime.toISOString(),
        },
      ],
      softConflicts: [],
    };
  }

  // Add buffer
  const bufferedStart = new Date(startTime.getTime() - BUFFER_MS);
  const bufferedEnd = new Date(endTime.getTime() + BUFFER_MS);

  const hardBlocks: CalendarConflict[] = [];
  const softConflicts: CalendarConflict[] = [];

  // Query all calendars
  const calendarIds = Object.values(CALENDAR_IDS);
  const calendarClient = createCalendarClient(params.calendarAuth.accessToken);

  const results = await Promise.allSettled(
    calendarIds.map((calId) =>
      calendarClient.events.list({
        calendarId: calId,
        timeMin: bufferedStart.toISOString(),
        timeMax: bufferedEnd.toISOString(),
        singleEvents: true,
        orderBy: "startTime",
        timeZone: TIMEZONE,
      }),
    ),
  );

  for (let i = 0; i < results.length; i++) {
    const result = results[i];
    if (result.status !== "fulfilled") continue;

    const calId = calendarIds[i];
    const calName = CALENDAR_NAMES[calId] ?? "Unknown";
    const events = result.value.data.items ?? [];

    for (const event of events) {
      if (!event.summary) continue;

      const parsed = parseEventPrefix(event.summary);
      const eventStart = event.start?.dateTime ?? event.start?.date ?? "";
      const eventEnd = event.end?.dateTime ?? event.end?.date ?? "";

      const conflict: CalendarConflict = {
        title: parsed.cleanTitle,
        calendar: calName,
        start: eventStart,
        end: eventEnd,
      };

      // FYI events are always ignored
      if (parsed.type === PrefixType.INFORMATIONAL) continue;

      // Nutrition and Workout calendars are always soft
      if (SOFT_CALENDARS.has(calId)) {
        softConflicts.push(conflict);
        continue;
      }

      // Apply prefix convention
      if (parsed.type === PrefixType.SOFT) {
        softConflicts.push(conflict);
      } else {
        hardBlocks.push(conflict);
      }
    }
  }

  return {
    available: hardBlocks.length === 0,
    hardBlocks,
    softConflicts,
  };
}

function createCalendarClient(accessToken: string): calendar_v3.Calendar {
  const { calendar } = require("@googleapis/calendar");
  return calendar({
    version: "v3",
    headers: { Authorization: `Bearer ${accessToken}` },
  });
}
```

```typescript
interface PreFilterOptions {
  allowedDomains?: string[];
  blockedDomains?: string[];
}

interface PreFilterOutput {
  action: PreFilterResult;
  reason: FilterReason | null;
}

const SKIP_CATEGORIES = new Set(["promotions", "social", "forums"]);

const SHIPPING_SENDERS = [
  "ups.com",
  "fedex.com",
  "usps.com",
  "dhl.com",
  "amazon.com",
  "ship-confirm@",
  "shipment-tracking@",
  "tracking@",
];

const SHIPPING_KEYWORDS = [
  "shipped",
  "has shipped",
  "out for delivery",
  "delivery",
  "tracking number",
  "your package",
  "order shipped",
  "shipment confirmation",
];

const BOUNCE_SENDERS = ["mailer-daemon@", "postmaster@"];

export function preFilterEmail(
  email: PreFilterInput,
  options: PreFilterOptions = {},
): PreFilterOutput {
  const fromLower = email.from.toLowerCase();
  const subjectLower = email.subject.toLowerCase();
  const fromDomain = fromLower.split("@")[1] ?? "";

  // Check allowlist first — if sender domain is allowed, always process
  if (options.allowedDomains?.some((d) => fromDomain === d.toLowerCase())) {
    return { action: PreFilterResult.PROCESS, reason: null };
  }

  // 1. Gmail category check
  if (email.category && SKIP_CATEGORIES.has(email.category.toLowerCase())) {
    return {
      action: PreFilterResult.SKIP,
      reason: FilterReason.GMAIL_CATEGORY,
    };
  }

  // 2. Blocked domains
  if (options.blockedDomains?.some((d) => fromDomain === d.toLowerCase())) {
    return { action: PreFilterResult.SKIP, reason: FilterReason.BLOCKLIST };
  }

  // 3. Mailing list headers
  const headers = Object.fromEntries(
    Object.entries(email.headers).map(([k, v]) => [k.toLowerCase(), v]),
  );
  if (headers["list-unsubscribe"] || headers["list-id"]) {
    return {
      action: PreFilterResult.SKIP,
      reason: FilterReason.MAILING_LIST,
    };
  }

  // 4. Bounce/delivery status
  if (BOUNCE_SENDERS.some((s) => fromLower.includes(s))) {
    return { action: PreFilterResult.SKIP, reason: FilterReason.BOUNCE };
  }

  // 5. Shipping/delivery emails
  const isShippingSender = SHIPPING_SENDERS.some((s) =>
    fromLower.includes(s),
  );
  const isShippingSubject = SHIPPING_KEYWORDS.some((k) =>
    subjectLower.includes(k),
  );
  if (isShippingSender || isShippingSubject) {
    return {
      action: PreFilterResult.CREATE_CALENDAR_EVENT,
      reason: FilterReason.SHIPPING,
    };
  }

  // 6. Gmail updates category (receipts, confirmations)
  if (email.category?.toLowerCase() === "updates") {
    return {
      action: PreFilterResult.BATCH_SUMMARY,
      reason: FilterReason.BATCH_SUMMARY,
    };
  }

  // 7. Everything else -> process with Claude
  return { action: PreFilterResult.PROCESS, reason: null };
}
```

```typescript
import { CALENDAR_IDS, TIMEZONE } from "./types";

const CARRIER_PATTERNS: Record<string, RegExp> = {
  Amazon: /amazon/i,
  FedEx: /fedex/i,
  UPS: /ups/i,
  USPS: /usps/i,
  DHL: /dhl/i,
};

const ITEM_PATTERNS = [
  /order of (.+?) has shipped/i,
  /shipped: (.+)/i,
  /your (.+?) has been shipped/i,
  /shipment.+?for (.+)/i,
];

export function parseShippingEmail(email: {
  subject: string;
  from: string;
}): string {
  const fromDomain = email.from.split("@")[1]?.replace(/^.*\.(?=\w+\.\w+$)/, "") ?? "Unknown";

  // Detect carrier
  let carrier: string | null = null;
  for (const [name, pattern] of Object.entries(CARRIER_PATTERNS)) {
    if (pattern.test(email.from) || pattern.test(email.subject)) {
      carrier = name;
      break;
    }
  }

  // Try to extract item name from subject
  for (const pattern of ITEM_PATTERNS) {
    const match = email.subject.match(pattern);
    if (match?.[1]) {
      const item = match[1].trim().slice(0, 50); // Cap length
      return `${item} (${carrier ?? fromDomain})`;
    }
  }

  // Fallback
  if (carrier) {
    return `Package (${carrier})`;
  }

  return `Order (${fromDomain})`;
}

    timeZone: TIMEZONE,
  }); // YYYY-MM-DD

  return {
```

```typescript
interface VentureInput {

  senderEmail: string;
  clientGroupVenture: Venture | null;

const INBOX_TO_VENTURE: Record<string, Venture> = {
  "nick@smartcollege.com": Venture.SMART_COLLEGE,
  "nick@growwithpraxis.com": Venture.PRAXIS,
};

const DOMAIN_TO_VENTURE: Record<string, Venture> = {
  "smartcollege.com": Venture.SMART_COLLEGE,
  "growwithpraxis.com": Venture.PRAXIS,
};

export function detectVenture(input: VentureInput): Venture {
  // 1. Check inbox
  const inboxVenture = INBOX_TO_VENTURE[input.inboxEmail.toLowerCase()];
  if (inboxVenture) return inboxVenture;

  // 2. Check sender domain
  const senderDomain = input.senderEmail.toLowerCase().split("@")[1];
  if (senderDomain) {
    const domainVenture = DOMAIN_TO_VENTURE[senderDomain];
    if (domainVenture) return domainVenture;
  }

  // 3. Check client group association
  if (input.clientGroupVenture) return input.clientGroupVenture;

  // 4. Default to Personal
  return Venture.PERSONAL;
}
```

```typescript
interface ClientInfo {
  email: string;
  name: string;
}

export function shouldAutoGroup(a: ClientInfo, b: ClientInfo): boolean {
  const domainA = a.email.toLowerCase().split("@")[1];
  const domainB = b.email.toLowerCase().split("@")[1];

  if (!domainA || !domainB || domainA !== domainB) return false;

  const lastNameA = extractLastName(a.name);
  const lastNameB = extractLastName(b.name);

  if (!lastNameA || !lastNameB) return false;

  return lastNameA.toLowerCase() === lastNameB.toLowerCase();
}

function extractLastName(fullName: string): string | null {
  const parts = fullName.trim().split(/\s+/);
  if (parts.length < 2) return null;
  return parts[parts.length - 1];
}
```

```typescript
// apps/web/utils/chief-of-staff/vip/detector.ts
import type { PrismaClient } from "@prisma/client";
import { getClientAppointments } from "../acuity/actions";
import { VIP_THRESHOLD } from "../types";

const CACHE_TTL_MS = 24 * 60 * 60 * 1000; // 24 hours

interface VipResult {
  isVip: boolean;
  bookingCount: number;
  groupName: string | null;
}

export async function checkVipStatus(
  clientEmail: string,
  prisma: PrismaClient,
): Promise<VipResult> {
  // Check cache first
  const cached = await prisma.vipCache.findUnique({
    where: { clientEmail },
  });

  if (cached && Date.now() - cached.lastChecked.getTime() < CACHE_TTL_MS) {
    let groupName: string | null = null;
    if (cached.clientGroupId) {
      const group = await prisma.clientGroup.findUnique({
        where: { id: cached.clientGroupId },
      });
      groupName = group?.primaryName ?? null;
    }
    return {
      isVip: cached.isVip,
      bookingCount: cached.bookingCount,
      groupName,
    };
  }

  // Find all emails in this client's group
  const emailsToCheck = [clientEmail];
  let groupId: string | null = null;
  let groupName: string | null = null;

  const membership = await prisma.clientGroupMember.findUnique({
    where: { email: clientEmail },
    include: {
      clientGroup: {
        include: { members: true },
      },
    },
  });

  if (membership) {
    groupId = membership.clientGroupId;
    groupName = membership.clientGroup.primaryName;
    for (const member of membership.clientGroup.members) {
      if (member.email !== clientEmail) {
        emailsToCheck.push(member.email);
      }
    }
  }

  // Query Acuity for all emails in group
  let totalBookings = 0;
  for (const email of emailsToCheck) {
    const appointments = await getClientAppointments(email);
    totalBookings += appointments.filter((a) => !a.canceled).length;
  }

  const isVip = totalBookings >= VIP_THRESHOLD;

  // Update cache
  await prisma.vipCache.upsert({
    where: { clientEmail },
    create: {
      clientEmail,
      clientGroupId: groupId,
      bookingCount: totalBookings,
      isVip,
      lastChecked: new Date(),
    },
    update: {
      clientGroupId: groupId,
      bookingCount: totalBookings,
      isVip,
      lastChecked: new Date(),
    },
  });

  return { isVip, bookingCount: totalBookings, groupName };
}
```

```typescript
    Date.now() - config.signatureLastFetched.getTime() < SIGNATURE_CACHE_TTL_MS
  ) {
    return config.signatureHtml;
  }

  // Fetch from Gmail API
  const response = await gmail.users.settings.sendAs.get({
    userId: "me",
    sendAsEmail: emailAddress,
  });

  const signature = response.data.signature ?? "";

  // Update cache
  if (config) {
    await prisma.chiefOfStaffConfig.update({
      where: { emailAccountId },
      data: {
        signatureHtml: signature,
        signatureLastFetched: new Date(),
      },
    });
  }

  return signature;
}

export function appendSignatureToBody(
  bodyHtml: string,
  signatureHtml: string,
): string {
  if (!signatureHtml) return bodyHtml;
  return `${bodyHtml}<br><br>${signatureHtml}`;
}
```

```typescript
import { AutonomyMode, CosCategory, Venture } from "./types";

interface PromptParams {
  venture: Venture;
  voiceTone: string;
  autonomyLevels: Record<CosCategory, AutonomyMode>;
  currentDateTime: string;

let basePromptCache: string | null = null;

function getBasePrompt(): string {
  if (basePromptCache) return basePromptCache;
  const promptPath = path.join(
    process.cwd(),
    "config",
    "system-prompt.md",
  );
  basePromptCache = fs.readFileSync(promptPath, "utf-8");
  return basePromptCache;
}

const VENTURE_NAMES: Record<Venture, string> = {
  [Venture.SMART_COLLEGE]: "Smart College",
  [Venture.PRAXIS]: "Praxis Education",
  [Venture.PERSONAL]: "Personal",
};

export function buildSystemPrompt(params: PromptParams): string {
  const base = getBasePrompt();

  const dynamicSections = [
    "---",
    "## Runtime Context (injected per email)",
    "",
    `**Current date/time:** ${params.currentDateTime} (America/Chicago)`,
    `**Responding as:** ${VENTURE_NAMES[params.venture]}`,
    `**Voice/Tone:** ${params.voiceTone}`,
    "",
    "### Autonomy Levels",
    "These define what you are allowed to do for each category:",
    "",
    ...Object.entries(params.autonomyLevels).map(
      ([cat, mode]) => `- **${cat}:** ${formatAutonomyMode(mode)}`,
    ),
    "",
    "### Day Protection Rules",
    "- Tuesday is a protected recovery day — never suggest tutoring on Tuesdays, no override allowed.",
    "- Friday is a protected non-tutoring day — only suggest Friday tutoring for VIP clients (5+ past bookings).",
    "",
    "### Autonomy Mode Definitions",
    "- **auto_handle:** Execute the action immediately (reschedule, book, send confirmation). Report what you did.",
    "- **draft_approve:** Create a Gmail draft and report it. Do NOT send — Nick will approve via Slack.",
    "- **flag_only:** Do NOT take any action or draft. Just report the situation with context and recommended action.",
  ];

  return `${base}\n\n${dynamicSections.join("\n")}`;
}

function formatAutonomyMode(mode: AutonomyMode): string {
  switch (mode) {
    case AutonomyMode.AUTO_HANDLE:
      return "Auto-Handle (execute and report)";
    case AutonomyMode.DRAFT_APPROVE:
      return "Draft + Approve (draft response, wait for Nick's approval)";
    case AutonomyMode.FLAG_ONLY:
      return "Flag Only (never auto-handle, always escalate)";
  }
}
```

```typescript
// apps/web/utils/chief-of-staff/engine.ts
import { generateText } from "ai";
import { createChiefOfStaffTools } from "./tools";
import { buildSystemPrompt } from "./system-prompt";
import {
  CosCategory,
  CosEngineResponse,
  AutonomyMode,
  EmailMetadata,
  Venture,
  TIMEZONE,
} from "./types";
import { getModel } from "@/utils/llms/model";
import { createScopedLogger } from "@/utils/logger";

const logger = createScopedLogger("chief-of-staff/engine");

interface ProcessEmailParams {
  email: EmailMetadata;
  venture: Venture;
  voiceTone: string;
  autonomyLevels: Record<CosCategory, AutonomyMode>;
  toolContext: Parameters<typeof createChiefOfStaffTools>[0];
}

export async function processEmailWithClaude(
  params: ProcessEmailParams,
): Promise<CosEngineResponse> {
  const { email, venture, voiceTone, autonomyLevels, toolContext } = params;

  const currentDateTime = new Date().toLocaleString("en-US", {
    timeZone: TIMEZONE,
    dateStyle: "full",
    timeStyle: "long",
  });

  const systemPrompt = buildSystemPrompt({
    venture,
    voiceTone,
    autonomyLevels,
    currentDateTime,
  });

  const tools = createChiefOfStaffTools(toolContext);

  const userMessage = formatEmailForClaude(email);

  const { model } = getModel(
    { aiProvider: "anthropic", aiModel: "claude-sonnet-4-6" } as any,
    "default",
  );

  const result = await generateText({
    model,
    system: systemPrompt,
    prompt: userMessage,
    tools,
    maxSteps: 10, // Allow multiple tool calls
  });

  // Parse the structured response from Claude's final text
  return parseEngineResponse(result.text, email);
}

function formatEmailForClaude(email: EmailMetadata): string {
  return [
    "Process this email:",
    "",
    `From: ${email.from}`,
    `To: ${email.to}`,
    `Subject: ${email.subject}`,
    `Date: ${email.date.toISOString()}`,
    `Thread ID: ${email.threadId}`,
    "",
    "Body:",
    email.body,
    "",
    "Respond with a JSON object containing: category, summary, action_taken, draft (or null), needs_approval, conflicts, is_vip, vip_group_name",
  ].join("\n");
}

function parseEngineResponse(
  text: string,
  email: EmailMetadata,
): CosEngineResponse {
  try {
    // Extract JSON from Claude's response (may be wrapped in markdown code blocks)
    const jsonMatch = text.match(/\{[\s\S]*\}/);
    if (!jsonMatch) {
      throw new Error("No JSON found in Claude response");
    }

    const parsed = JSON.parse(jsonMatch[0]);

    return {
      category: parsed.category as CosCategory,
      summary: parsed.summary ?? "No summary provided",
      actionTaken: parsed.action_taken ?? null,
      draft: parsed.draft
        ? {
            to: parsed.draft.to,
            subject: parsed.draft.subject,
            body: parsed.draft.body,
            gmailDraftId: parsed.draft.gmail_draft_id ?? parsed.draft.draftId ?? "",
            gmailThreadId: parsed.draft.thread_id ?? email.threadId,
          }
        : null,
      needsApproval: parsed.needs_approval ?? true,
      conflicts: parsed.conflicts ?? [],
      isVip: parsed.is_vip ?? false,
      vipGroupName: parsed.vip_group_name ?? null,
    };
  } catch (error) {
    logger.error("Failed to parse Claude response", { error, text });
    // Fallback: treat as urgent/flag-only
    return {
      category: CosCategory.URGENT,
      summary: `Failed to process: ${email.subject}`,
      actionTaken: null,
      draft: null,
      needsApproval: true,
      conflicts: [],
      isVip: false,
      vipGroupName: null,
    };
  }
}
```

```typescript
import { BATCH_WINDOW_MS } from "../types";

interface PendingPost {
  timestamp: number;
  postFn: (threadTs?: string) => Promise<string>;
  fallbackText: string;
}

let pendingPosts: PendingPost[] = [];
let batchTimer: ReturnType<typeof setTimeout> | null = null;

export function addToBatch(post: PendingPost): void {
  pendingPosts.push(post);

  if (batchTimer) clearTimeout(batchTimer);
  batchTimer = setTimeout(flushBatch, BATCH_WINDOW_MS);
}

export function shouldBatch(): boolean {
  return pendingPosts.length >= 3;
}

export async function flushBatch(): Promise<void> {
  if (batchTimer) {
    clearTimeout(batchTimer);
    batchTimer = null;
  }

  const posts = [...pendingPosts];
  pendingPosts = [];

  if (posts.length === 0) return;

  if (posts.length < 3) {
    // Post individually (no batch parent)
    for (const post of posts) {
      await post.postFn();
    }
    return;
  }

  // Post as batch: parent message + thread replies
  // The parent message is posted by the caller (webhook process)
  // Individual posts go as thread replies
  // This is handled in the webhook processor, not here
}

export function getPendingCount(): number {
  return pendingPosts.length;
}

export function clearPending(): void {
  pendingPosts = [];
  if (batchTimer) {
    clearTimeout(batchTimer);
    batchTimer = null;
  }
}
```

```typescript
// apps/web/app/api/chief-of-staff/slack/interactions/route.ts
import { NextResponse } from "next/server";
import crypto from "crypto";
import prisma from "@/utils/prisma";
import {
  handleApprove,
  handleEdit,
  handleEditSubmit,
  handleReject,
} from "@/utils/chief-of-staff/slack/actions";
import { createScopedLogger } from "@/utils/logger";

const logger = createScopedLogger("chief-of-staff/slack-interactions");

export async function POST(request: Request) {
  // Verify Slack signing secret
  const signature = request.headers.get("x-slack-signature");
  const timestamp = request.headers.get("x-slack-request-timestamp");
  const rawBody = await request.text();

  if (!verifySlackSignature(signature, timestamp, rawBody)) {
    return NextResponse.json({ error: "Invalid signature" }, { status: 401 });
  }

  const payload = JSON.parse(
    new URLSearchParams(rawBody).get("payload") ?? "{}",
  );

  const channelId = process.env.CHIEF_OF_STAFF_SLACK_CHANNEL_ID!;

  // Get Slack access token from messaging channel
  const messagingChannel = await prisma.messagingChannel.findFirst({
    where: { channelId, provider: "slack" },
  });

  if (!messagingChannel?.accessToken) {
    logger.error("No Slack access token found", { channelId });
    return NextResponse.json({ error: "No token" }, { status: 500 });
  }

  const slackAccessToken = messagingChannel.accessToken;

  // Route based on interaction type
  if (payload.type === "block_actions") {
    const action = payload.actions?.[0];
    const messageTs = payload.message?.ts;

    if (!action || !messageTs) {
      return NextResponse.json({ ok: true });
    }

    const getGmailClient = async (emailAccountId: string) => {
      const { getGmailClientForAccount } = require("@/utils/gmail/client");
      return getGmailClientForAccount(emailAccountId);
    };

    switch (action.action_id) {
      case "cos_approve":
        await handleApprove({
          slackMessageTs: messageTs,
          channelId,
          slackAccessToken,
          prisma,
          getGmailClient,
        });
        break;

      case "cos_edit":
        await handleEdit({
          slackMessageTs: messageTs,
          triggerId: payload.trigger_id,
          slackAccessToken,
          prisma,
        });
        break;

      case "cos_reject":
        await handleReject({
          slackMessageTs: messageTs,
          channelId,
          slackAccessToken,
          prisma,
          getGmailClient,
        });
        break;
    }
  }

  // Handle modal submissions (edit modal)
  if (
    payload.type === "view_submission" &&
    payload.view?.callback_id === "cos_edit_modal"
  ) {
    const slackMessageTs = payload.view.private_metadata;
    const newBody =
      payload.view.state.values.draft_body.draft_body_input.value;

    const getGmailClient = async (emailAccountId: string) => {
      const { getGmailClientForAccount } = require("@/utils/gmail/client");
      return getGmailClientForAccount(emailAccountId);
    };

    await handleEditSubmit({
      slackMessageTs,
      newBody,
      channelId,
      slackAccessToken,
      prisma,
      getGmailClient,
    });
  }

  return NextResponse.json({ ok: true });
}

function verifySlackSignature(
  signature: string | null,
  timestamp: string | null,
  body: string,
): boolean {
  const signingSecret = process.env.SLACK_SIGNING_SECRET;
  if (!signingSecret || !signature || !timestamp) return false;

  // Check timestamp is within 5 minutes
  const time = Math.floor(Date.now() / 1000);
  if (Math.abs(time - parseInt(timestamp)) > 300) return false;

  const sigBasestring = `v0:${timestamp}:${body}`;
  const mySignature = `v0=${crypto
    .createHmac("sha256", signingSecret)
    .update(sigBasestring)
    .digest("hex")}`;

  return crypto.timingSafeEqual(
    Buffer.from(mySignature),
    Buffer.from(signature),
  );
}
```

```typescript
async function processOneEmail(params: {
  messageId: string;
  emailAccount: any;
  gmail: any;
  autonomyLevels: Record<CosCategory, AutonomyMode>;
  slackAccessToken: string;
  slackChannelId: string;
  logger: Logger;
}): Promise<void> {
  const {
    messageId,
    emailAccount,
    gmail,
    autonomyLevels,
    slackAccessToken,
    slackChannelId,
    logger,
  } = params;

  // Fetch full message
  const message = await getMessage(messageId, gmail);
  const parsed = parseGmailToEmailMetadata(message, emailAccount.email);

  // Pre-filter
  const filterResult = preFilterEmail({
    from: parsed.from,
    subject: parsed.subject,
    labels: parsed.labels,
    category: parsed.category,
    headers: parsed.headers,
  });

  if (filterResult.action === PreFilterResult.SKIP) {
    await prisma.cosFilteredEmail.upsert({
      where: { messageId },
      create: {
        messageId,
        emailAccountId: emailAccount.id,
        sender: parsed.from,
        filterReason: filterResult.reason!,
      },
      update: {},
    });
    return;
  }

  if (filterResult.action === PreFilterResult.CREATE_CALENDAR_EVENT) {
    const itemDesc = parseShippingEmail({
      subject: parsed.subject,
      from: parsed.from,
    });
    // Create calendar event (implementation uses Google Calendar API)
    // Log to ShippingEvent and FilteredEmail tables
    await prisma.cosFilteredEmail.upsert({
      where: { messageId },
      create: {
        messageId,
        emailAccountId: emailAccount.id,
        sender: parsed.from,
        filterReason: "shipping",
      },
      update: {},
    });
    logger.info("Created shipping calendar event", { itemDesc });
    return;
  }

  if (filterResult.action === PreFilterResult.BATCH_SUMMARY) {
    await prisma.cosFilteredEmail.upsert({
      where: { messageId },
      create: {
        messageId,
        emailAccountId: emailAccount.id,
        sender: parsed.from,
        filterReason: "batch_summary",
      },
      update: {},
    });
    return;
  }

  // Process with Claude
  const processedEmail = await prisma.processedEmail.create({
    data: {
      messageId,
      threadId: parsed.threadId,
      emailAccountId: emailAccount.id,
      status: ProcessedEmailStatus.PROCESSING,
    },
  });

  try {
    // Detect venture
    const clientGroupVenture = null; // TODO: look up from ClientGroupMember
    const venture = detectVenture({
      inboxEmail: emailAccount.email,
      senderEmail: parsed.from,
      clientGroupVenture,
    });

    const voiceProfile = getVoiceToneProfile(venture);
    const voiceTone = formatVoiceToneForPrompt(voiceProfile);

    // Run Claude engine
    const response = await processEmailWithClaude({
      email: parsed,
      venture,
      voiceTone,
      autonomyLevels,
      toolContext: {
        emailAccountId: emailAccount.id,
        emailAddress: emailAccount.email,
        gmail,
        prisma,
        calendarAuth: {
          accessToken: emailAccount.account.access_token,
          refreshToken: emailAccount.account.refresh_token,
        },
      },
    });

    // Determine action based on autonomy level
    const mode = autonomyLevels[response.category] ?? AutonomyMode.DRAFT_APPROVE;

    if (mode === AutonomyMode.FLAG_ONLY) {
      // Post flag-only message (no buttons, no draft)
      await postToChiefOfStaff({
        accessToken: slackAccessToken,
        channelId: slackChannelId,
        blocks: buildFlagOnlyMessage({
          fromEmail: parsed.from,
          subject: parsed.subject,
          summary: response.summary,
          venture,
        }),
        text: `Urgent: ${parsed.subject}`,
      });
    } else if (mode === AutonomyMode.AUTO_HANDLE) {
      // Post auto-handle notification (no buttons)
      await postToChiefOfStaff({
        accessToken: slackAccessToken,
        channelId: slackChannelId,
        blocks: buildAutoHandleMessage({
          summary: response.summary,
          actionTaken: response.actionTaken,
        }),
        text: `Auto-handled: ${response.summary}`,
      });
    } else {
      // Draft + Approve: post with buttons
      const slackTs = await postToChiefOfStaff({
        accessToken: slackAccessToken,
        channelId: slackChannelId,
        blocks: buildApprovalMessage({
          response,
          fromEmail: parsed.from,
          subject: parsed.subject,
          venture,
        }),
        text: `New email from ${parsed.from}: ${parsed.subject}`,
      });

      // Save pending draft
      if (response.draft) {
        await prisma.cosPendingDraft.create({
          data: {
            slackMessageTs: slackTs,
            slackChannelId,
            gmailDraftId: response.draft.gmailDraftId,
            gmailThreadId: response.draft.gmailThreadId,
            emailAccountId: emailAccount.id,
            toAddress: response.draft.to,
            subject: response.draft.subject,
            bodyHtml: response.draft.body,
            category: response.category,
            claudeResponse: response as any,
            processedEmailId: processedEmail.id,
          },
        });
      }
    }

    // Update processed email
    await prisma.processedEmail.update({
      where: { id: processedEmail.id },
      data: {
        status: ProcessedEmailStatus.COMPLETED,
        category: response.category,
        actionTaken: response.actionTaken,
      },
    });
  } catch (error) {
    await prisma.processedEmail.update({
      where: { id: processedEmail.id },
      data: {
        status: ProcessedEmailStatus.FAILED,
        failedStage: "processing",
        retryCount: { increment: 1 },
      },
    });
    throw error;
  }
}

function parseGmailToEmailMetadata(
  message: any,
  inboxEmail: string,
): EmailMetadata {
  const headers: Record<string, string> = {};
  for (const header of message.payload?.headers ?? []) {
    headers[header.name.toLowerCase()] = header.value;
  }

  return {
    messageId: message.id,
    threadId: message.threadId,
    from: headers.from ?? "",
    to: headers.to ?? inboxEmail,
    subject: headers.subject ?? "(no subject)",
    date: new Date(headers.date ?? Date.now()),
    labels: message.labelIds ?? [],
    category: detectGmailCategory(message.labelIds ?? []),
    headers,
    snippet: message.snippet ?? "",
    body: extractBody(message),
  };
}

function detectGmailCategory(labelIds: string[]): string | null {
  if (labelIds.includes("CATEGORY_PROMOTIONS")) return "promotions";
  if (labelIds.includes("CATEGORY_SOCIAL")) return "social";
  if (labelIds.includes("CATEGORY_UPDATES")) return "updates";
  if (labelIds.includes("CATEGORY_FORUMS")) return "forums";
  return null;
}

function extractBody(message: any): string {
  // Extract text/plain or text/html body from Gmail message payload
  const payload = message.payload;
  if (!payload) return "";

  if (payload.body?.data) {
    return Buffer.from(payload.body.data, "base64").toString();
  }

  // Check parts recursively
  for (const part of payload.parts ?? []) {
    if (part.mimeType === "text/plain" && part.body?.data) {
      return Buffer.from(part.body.data, "base64").toString();
    }
    if (part.mimeType === "text/html" && part.body?.data) {
      return Buffer.from(part.body.data, "base64").toString();
    }
    // Recurse into multipart
    if (part.parts) {
      for (const subPart of part.parts) {
        if (subPart.body?.data) {
          return Buffer.from(subPart.body.data, "base64").toString();
        }
      }
    }
  }

  return message.snippet ?? "";
}
```

```typescript
import { createScopedLogger } from "@/utils/logger";

const logger = createScopedLogger("chief-of-staff/jobs/retry-failed");

export async function retryFailedEmails(): Promise<{
  retried: number;
  deadLettered: number;
}> {
  const failedEmails = await prisma.processedEmail.findMany({
    where: {
      status: ProcessedEmailStatus.FAILED,
      retryCount: { lt: MAX_RETRY_COUNT },
    },
    orderBy: { createdAt: "asc" },
    take: 10,
  });

  let retried = 0;
  let deadLettered = 0;

  for (const email of failedEmails) {
    if (email.retryCount >= MAX_RETRY_COUNT - 1) {
      // Max retries reached — dead letter
      await prisma.processedEmail.update({
        where: { id: email.id },
        data: { status: ProcessedEmailStatus.DEAD_LETTER },
      });
      deadLettered++;

      // TODO: Post Slack alert about dead-lettered email
      logger.error("Email dead-lettered after max retries", {
        messageId: email.messageId,
        failedStage: email.failedStage,
      });
    } else {
      // Retry by re-triggering the processing pipeline
      // For now, increment retry count — the actual retry logic
      // will be wired in when we integrate with the webhook processor
      await prisma.processedEmail.update({
        where: { id: email.id },
        data: { retryCount: { increment: 1 } },
      });
      retried++;
    }
  }

  return { retried, deadLettered };
}
```

```typescript
// apps/web/app/api/chief-of-staff/cron/route.ts
import { NextResponse } from "next/server";
import { retryFailedEmails } from "@/utils/chief-of-staff/jobs/retry-failed";
import { postBatchSummary } from "@/utils/chief-of-staff/jobs/batch-summary";
import prisma from "@/utils/prisma";
import { createScopedLogger } from "@/utils/logger";

const logger = createScopedLogger("chief-of-staff/cron");

export async function GET(request: Request) {
  const url = new URL(request.url);
  const jobType = url.searchParams.get("type");

  // Simple auth — verify a shared secret
  const authHeader = request.headers.get("authorization");
  if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  switch (jobType) {
    case "retry-failed": {
      const result = await retryFailedEmails();
      return NextResponse.json(result);
    }

    case "batch-summary": {
      const channelId = process.env.CHIEF_OF_STAFF_SLACK_CHANNEL_ID;
      const messagingChannel = await prisma.messagingChannel.findFirst({
        where: { channelId, provider: "slack" },
      });

      if (!messagingChannel?.accessToken || !channelId) {
        return NextResponse.json(
          { error: "Missing Slack config" },
          { status: 500 },
        );
      }

      const count = await postBatchSummary({
        slackAccessToken: messagingChannel.accessToken,
        channelId,
      });
      return NextResponse.json({ summarized: count });
    }

    default:
      return NextResponse.json(
        { error: `Unknown job type: ${jobType}` },
        { status: 400 },
      );
  }
}
```
