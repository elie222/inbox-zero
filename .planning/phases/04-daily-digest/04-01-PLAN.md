```ts
import { createScopedLogger } from "@/utils/logger";

const baseLogger = createScopedLogger("digest/instrumentation");

/**
 * Phase 4 anomaly A2 instrumentation.
 *
 * Per Phase 3 verification: ~12% of DIGEST ExecutedActions do not produce a corresponding
 * DigestItem row. Likely cause is silent failure in the BullMQ enqueue path.
 *
 * Call sites (DIGEST action handler and DigestItem creation site) must wrap their work
 * with `traceDigestEnqueue` so production logs let us correlate ExecutedAction → DigestItem.
 */
export async function traceDigestEnqueue<T>(
  ctx: { executedActionId: string; emailAccountId: string; messageId: string; ruleName: string },
  fn: () => Promise<T>,
): Promise<T> {
  const logger = baseLogger.with(ctx);
  logger.info("digest.enqueue.start");
  try {
    const result = await fn();
    logger.info("digest.enqueue.success");
    return result;
  } catch (err) {
    logger.error("digest.enqueue.failure", { error: err instanceof Error ? err.message : String(err) });
    throw err;
  }
}

/**
 * Reconciliation query: count DIGEST ExecutedActions that have NO corresponding DigestItem
 * within `windowMinutes`. Run from a manual ops script during Phase 4 first-week monitoring;
 * NOT wired to a cron in v1 (per RESEARCH.md §"Failure Visibility" — defer to v2).
 *
 * Returns: number of orphan DIGEST ExecutedActions in the window.
 */
export async function countOrphanDigestActions(
  prisma: import("@prisma/client").PrismaClient,
  windowMinutes = 60,
): Promise<number> {
  const since = new Date(Date.now() - windowMinutes * 60_000);
  const result = await prisma.$queryRaw<{ count: bigint }[]>`
    SELECT COUNT(*)::bigint AS count
    FROM "ExecutedAction" ea
    WHERE ea.type = 'DIGEST'
      AND ea."createdAt" >= ${since}
      AND NOT EXISTS (
        SELECT 1 FROM "DigestItem" di
        WHERE di."actionId" = ea.id
      )
  `;
  return Number(result[0]?.count ?? 0n);
}
```
