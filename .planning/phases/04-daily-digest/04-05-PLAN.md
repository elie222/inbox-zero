```ts
/**
 * Returns today's ET calendar date as a Date at UTC midnight.
 *
 * Example: at 2026-05-04T03:00:00Z (which is 2026-05-03 23:00 ET) → returns Date for 2026-05-03 00:00:00Z.
 *
 * Used by digest-send idempotency: `(emailAccountId, date)` unique key prevents double-sends within
 * the same ET day. DST-safe because Intl handles the offset.
 */
export function getTodayET(): Date {
  const ymd = new Intl.DateTimeFormat("en-CA", {
    timeZone: "America/New_York",
    year: "numeric",
    month: "2-digit",
    day: "2-digit",
  }).format(new Date());
  // ymd format: "YYYY-MM-DD" → Date at UTC midnight
  return new Date(`${ymd}T00:00:00.000Z`);
}

export function formatTodayHumanET(): string {
  return new Intl.DateTimeFormat("en-US", {
    timeZone: "America/New_York",
    weekday: "long",
    month: "long",
    day: "numeric",
    year: "numeric",
  }).format(new Date());
}
```

```ts
import type { PrismaClient, DigestSend } from "@prisma/client";

export async function digestAlreadySentToday(
  prisma: PrismaClient,
  emailAccountId: string,
  todayET: Date,
): Promise<DigestSend | null> {
  return prisma.digestSend.findUnique({
    where: { emailAccountId_date: { emailAccountId, date: todayET } },
  });
}

export type RecordDigestSendInput = {
  emailAccountId: string;
  date: Date;
  sentAt: Date;
  resendMessageId: string | null;
  itemCount: number;
  inputTokens?: number;
  outputTokens?: number;
  modelUsed?: string;
  narrativeSnapshot?: string;
  digestIds: string[];
};

/**
 * Build the Prisma create-args object for DigestSend. Caller composes this into a $transaction
 * so the SENT-commit + DigestSend row + DigestItem redact happen atomically.
 */
export function buildDigestSendCreate(input: RecordDigestSendInput) {
  return {
    data: {
      emailAccountId: input.emailAccountId,
      date: input.date,
      sentAt: input.sentAt,
      resendMessageId: input.resendMessageId,
      itemCount: input.itemCount,
      inputTokens: input.inputTokens ?? null,
      outputTokens: input.outputTokens ?? null,
      modelUsed: input.modelUsed ?? null,
      narrativeSnapshot: input.narrativeSnapshot?.slice(0, 4000) ?? null,
      digestIds: input.digestIds,
    },
  };
}
```

```ts
import { prisma } from "@/utils/prisma";
import { DigestStatus } from "@prisma/client";
import { sleep } from "@/utils/sleep";
import { captureException } from "@/utils/error";
import type { Logger } from "@/utils/logger";
import { env } from "@/env";
import { sendDigestV2Email } from "@inboxzero/resend";
import type { DigestV2Props, AutoFiledGroup } from "../../../packages/resend/emails/digest-v2"; // adjust path to project alias
import { getTodayET, formatTodayHumanET } from "./today-et";
import { digestAlreadySentToday, buildDigestSendCreate } from "./digest-send";
import { generateDigestContent } from "@/utils/ai/digest/generate-digest-content";
// existing helpers — verify exact import paths from analog:
import { getEmailProvider } from "@/utils/email-provider";          // adjust to actual location
import { emailToContentForAI } from "@/utils/email-content";        // adjust to actual location

type RuleName = "Urgent" | "Uncertain" | "Receipts" | "Newsletters" | "Marketing" | "Notifications";

const BUCKET_BY_RULE: Record<string, RuleName> = {
  Urgent: "Urgent",
  Uncertain: "Uncertain",
  Receipts: "Receipts",
  Newsletters: "Newsletters",
  Marketing: "Marketing",
  Notifications: "Notifications",
};

const AUTO_FILED_TITLES: Record<"receipts" | "newsletters" | "marketing" | "notifications", string> = {
  receipts: "Receipts",
  newsletters: "Newsletters",
  marketing: "Marketing",
  notifications: "Notifications",
};

export async function runDailyDigest(logger: Logger) {
  const todayET = getTodayET();
  const todayHuman = formatTodayHumanET();

  // 1. Find all email accounts with un-SENT digests pending. Single-tenant in v1, but query is generic.
  const accounts = await prisma.emailAccount.findMany({
    where: { digests: { some: { status: { not: DigestStatus.SENT } } } },
    include: { user: true },
  });

  const results: Array<{ emailAccountId: string; sent: boolean; reason?: string; itemCount?: number; resendMessageId?: string | null }> = [];

  for (const account of accounts) {
    const scoped = logger.with({ emailAccountId: account.id, todayET: todayET.toISOString() });

    // 1a. Idempotency short-circuit
    const existing = await digestAlreadySentToday(prisma, account.id, todayET);
    if (existing) {
      scoped.info("digest.skip.alreadySent", { existingId: existing.id });
      results.push({ emailAccountId: account.id, sent: false, reason: "already-sent-today" });
      continue;
    }

    // 1b. Fetch un-SENT digests + items + rule names
    const pendingDigests = await prisma.digest.findMany({
      where: { emailAccountId: account.id, status: DigestStatus.PENDING },
      include: {
        items: {
          include: {
            action: {
              include: { executedRule: { include: { rule: { select: { name: true } } } } },
            },
          },
        },
      },
    });

    if (pendingDigests.length === 0) {
      scoped.info("digest.skip.nothingPending");
      results.push({ emailAccountId: account.id, sent: false, reason: "nothing-pending" });
      continue;
    }

    // 1c. Transition to PROCESSING
    await prisma.digest.updateMany({
      where: { id: { in: pendingDigests.map((d) => d.id) } },
      data: { status: DigestStatus.PROCESSING },
    });

    try {
      // 2. Collect messageIds, fetch Gmail bodies in batches of 100 with 2s pacing
      const allItems = pendingDigests.flatMap((d) => d.items);
      const messageIds = allItems.map((i) => i.messageId).filter(Boolean) as string[];

      const emailProvider = await getEmailProvider(account.id);
      const messages: Awaited<ReturnType<typeof emailProvider.getMessagesBatch>> = [];
      const batchSize = 100;
      for (let i = 0; i < messageIds.length; i += batchSize) {
        const batch = messageIds.slice(i, i + batchSize);
        const batchResults = await emailProvider.getMessagesBatch(batch);
        messages.push(...batchResults);
        if (i + batchSize < messageIds.length) await sleep(2000);
      }
      const messageMap = new Map(messages.map((m) => [m.id, m]));

      // 3. Bucket items by rule name
      const buckets = {
        urgent: [] as Array<{ messageId: string; subject: string; from: string; body: string; itemId: string }>,
        uncertain: [] as Array<{ messageId: string; subject: string; from: string; body: string; itemId: string }>,
        receipts: [] as Array<{ messageId: string; subject: string; from: string; body: string; itemId: string }>,
        newsletters: [] as Array<{ messageId: string; subject: string; from: string; body: string; itemId: string }>,
        marketing: [] as Array<{ messageId: string; subject: string; from: string; body: string; itemId: string }>,
        notifications: [] as Array<{ messageId: string; subject: string; from: string; body: string; itemId: string }>,
      };

      for (const item of allItems) {
        const ruleName = item.action?.executedRule?.rule?.name;
        const bucket = ruleName ? BUCKET_BY_RULE[ruleName] : undefined;
        if (!bucket) continue;
        const msg = messageMap.get(item.messageId);
        if (!msg) continue;
        buckets[bucket.toLowerCase() as keyof typeof buckets].push({
          messageId: item.messageId,
          subject: msg.subject ?? "(no subject)",
          from: msg.from ?? "(unknown sender)",
          body: emailToContentForAI(msg) ?? "",
          itemId: item.id,
        });
      }

      // 4. Sonnet batched call
      const content = await generateDigestContent({
        emailAccount: account as any,    // use EmailAccountWithAI shape if account.user already loaded
        todayDate: todayHuman,
        bucketed: {
          urgent: buckets.urgent,
          uncertain: buckets.uncertain,
          receipts: buckets.receipts,
          newsletters: buckets.newsletters,
          marketing: buckets.marketing,
          notifications: buckets.notifications,
        },
      });

      // 5. Merge Sonnet output with messageMap → DigestV2Props
      const reviewBase = "https://inbox.tdfurn.com/uncertain";
      const buildActionItems = (bucket: typeof buckets.urgent, sonnetItems: Array<{ messageId: string; summary: string }>) =>
        sonnetItems
          .map((s) => {
            const src = bucket.find((b) => b.messageId === s.messageId);
            if (!src) return null;
            const senderMatch = /^(.*?)(?:\s*<([^>]+)>)?$/.exec(src.from);
            return {
              subject: src.subject,
              senderName: senderMatch?.[1]?.trim() || src.from,
              senderEmail: senderMatch?.[2],
              summary: s.summary,
              reviewUrl: `${reviewBase}/${src.itemId}`,
            };
          })
          .filter(Boolean) as DigestV2Props["urgent"];

      const autoFiled: AutoFiledGroup[] = (["receipts", "newsletters", "marketing", "notifications"] as const).map((cat) => ({
        category: cat,
        title: AUTO_FILED_TITLES[cat],
        emailCount: buckets[cat].length,
        clusterCount: content.autoFiled[cat].length,
        rows: content.autoFiled[cat].map((c) => ({ label: c.label, summary: c.summary })),
      }));

      const props: DigestV2Props = {
        baseUrl: "https://inbox.tdfurn.com",
        date: todayHuman,
        sentTime: new Intl.DateTimeFormat("en-US", { timeZone: "America/New_York", hour: "numeric", minute: "2-digit", hour12: true }).format(new Date()),
        narrativeGreeting: content.narrativeGreeting,
        narrativeBody: content.narrativeBody,
        urgent: buildActionItems(buckets.urgent, content.urgent),
        uncertain: buildActionItems(buckets.uncertain, content.uncertain),
        autoFiled,
      };

      // 6. Send
      const subject = `Daily digest · ${todayHuman}`;   // D-15
      const sendResult = await sendDigestV2Email({
        from: env.RESEND_FROM_EMAIL,
        to: account.email,
        emailProps: props,
        subject,
      });

      // 7. Atomic SENT commit + DigestSend create + DigestItem redact
      const totalItems = Object.values(buckets).reduce((n, b) => n + b.length, 0);
      await prisma.$transaction([
        prisma.digest.updateMany({
          where: { id: { in: pendingDigests.map((d) => d.id) } },
          data: { status: DigestStatus.SENT, sentAt: new Date() },
        }),
        prisma.digestItem.updateMany({
          where: { digestId: { in: pendingDigests.map((d) => d.id) } },
          data: { content: "[REDACTED]" },
        }),
        prisma.digestSend.create(buildDigestSendCreate({
          emailAccountId: account.id,
          date: todayET,
          sentAt: new Date(),
          resendMessageId: sendResult.id,
          itemCount: totalItems,
          modelUsed: "claude-sonnet-4-6",
          narrativeSnapshot: content.narrativeBody,
          digestIds: pendingDigests.map((d) => d.id),
        })),
      ]);

      scoped.info("digest.send.success", { itemCount: totalItems, resendMessageId: sendResult.id });
      results.push({ emailAccountId: account.id, sent: true, itemCount: totalItems, resendMessageId: sendResult.id });
    } catch (error) {
      // Rollback: PROCESSING → FAILED so a retry next morning can pick them up by re-processing
      await prisma.digest.updateMany({
        where: { id: { in: pendingDigests.map((d) => d.id) } },
        data: { status: DigestStatus.FAILED },
      });
      scoped.error("digest.send.failure", { error: error instanceof Error ? error.message : String(error) });
      captureException(error);
      results.push({ emailAccountId: account.id, sent: false, reason: "error" });
      // Continue to next account rather than throwing — multi-account safety
    }
  }

  return { processedAccounts: accounts.length, results };
}
```

```ts
import { NextResponse } from "next/server";
import { withError } from "@/utils/middleware";
import { hasCronSecret, hasPostCronSecret } from "@/utils/cron";
import { captureException } from "@/utils/error";
import { runDailyDigest } from "@/utils/digest/run-daily-digest";

export const maxDuration = 300;

export const GET = withError("cron/digest", async (request) => {
  if (!hasCronSecret(request)) {
    captureException(new Error("Unauthorized request: api/cron/digest"));
    return new Response("Unauthorized", { status: 401 });
  }
  const result = await runDailyDigest(request.logger);
  return NextResponse.json(result);
});

export const POST = withError("cron/digest", async (request) => {

    return new Response("Unauthorized", { status: 401 });
```
